    date=datetime.date(tomap.YEAR[i],tomap.MONTH[i],tomap.DAY_OF_MONTH[i])
    ordinalday=date.toordinal()
    ordinalhour=(ordinalday-1)*24+tomap.dep_hour_round
    ordinaldays.append(ordinalday)
    ordinalhours.append(ordinalhour)
    
tomap["OrdinalDay"]=ordinaldays

#This is the cell with the dynamical visualization.
#With all modules loaded and run (it cannot run inline since ipython notebook has trouble with inline anomations)
#it will produce a dynamic animation which shows airport locations and uses point size to represent the number of flights
#in a time step and color to represent the rate of delays.
map = Basemap(projection='merc',
    resolution = 'h', area_thresh = 1000.0,
    llcrnrlon= -177.365570, llcrnrlat=17.657787,
    urcrnrlon=-49.572601, urcrnrlat=71.860726)
 
map.drawcoastlines()
map.drawcountries()
map.fillcontinents(color='#99ff33')
#map.bluemarble()
map.drawmapboundary()

lons =  0
lats = 0
x, y = map(lons, lats)
point = map.scatter(x, y, marker='o' ,s=10, cmap=cm.hot)

#Number of time steps in the data subset
numframes= len(dayset)

#This needs to be a list in which each element is a list of the mean delay/delay-factors for
#every airport in a time step. Thus, it is a list of lists or will function as such.
#We will instead access this information within the animation loop.
#for some reason, it freaks out if you take this out, even though we don't use it
#so i'll leave it here for now
colors = .5

def init():
    return point,

# animation function.  This is called sequentially
def animate(i, data, thepoint):
    lonstemp= list(byday.loc[i].lonO['mean'])
    latstemp = list(byday.loc[i].latO['mean'])
    xtemp,ytemp = map(lonstemp,latstemp)
    thepoint.set_offsets(zip(xtemp,ytemp))
    
    #This is to set color. Corresponds to how late.
    normfactor=np.max(np.array(byday.DEP_DELAY_NEW['mean']))
    thepoint.set_array(np.array(byday.loc[i].DEP_DELAY_NEW['mean'])/normfactor)

    
    #Make sure to normalize values
    normfactor2=np.max(np.array(byday.DEP_DELAY_NEW['count']))
    thepoint._sizes = 25.*np.array(byday.loc[i].DEP_DELAY_NEW['count'])/normfactor
    return thepoint,

# call the animator.  blit=True means only re-draw the parts that have changed.
anim = animation.FuncAnimation(plt.gcf(), animate, init_func=init,
                               frames=dayset, fargs=(colors,point), interval=500, blit=True)

plt.show()
